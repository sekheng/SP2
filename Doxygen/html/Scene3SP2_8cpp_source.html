390 {

393 
394  glDisable(GL_DEPTH_TEST);

399  glActiveTexture(GL_TEXTURE0);

403  {
404  Mtx44 characterSpacing;

407  Mtx44 MVP = projectionStack.Top() * viewStack.Top() * modelStack.Top() * characterSpacing;
