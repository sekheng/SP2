728 void sceneSP2::renderMesh(Mesh *mesh, bool enableLight)
729 {
730  Mtx44 MVP, modelView, modelView_inverse_transpose;
731  MVP = projectionStack.Top() * viewStack.Top() * modelStack.Top();
732  glUniformMatrix4fv(m_parameters[U_MVP], 1, GL_FALSE, &MVP.a[0]);
733  modelView = viewStack.Top() * modelStack.Top();
734  glUniformMatrix4fv(m_parameters[U_MODELVIEW], 1, GL_FALSE, &modelView.a[0]);
735 
736  if (enableLight)
737  {
738  glUniform1i(m_parameters[U_LIGHTENABLED], 1);
739  modelView_inverse_transpose = modelView.GetInverse().GetTranspose();
740  glUniformMatrix4fv(m_parameters[U_MODELVIEW_INVERSE_TRANSPOSE], 1, GL_FALSE, &modelView_inverse_transpose.a[0]);
741  //load material
742  glUniform3fv(m_parameters[U_MATERIAL_AMBIENT], 1, &mesh->material.kAmbient.r);
743  glUniform3fv(m_parameters[U_MATERIAL_DIFFUSE], 1, &mesh->material.kDiffuse.r);
744  glUniform3fv(m_parameters[U_MATERIAL_SPECULAR], 1, &mesh->material.kSpecular.r);
745  glUniform1f(m_parameters[U_MATERIAL_SHININESS], mesh->material.kShininess);
746  }
747  else
748  {
749  glUniform1i(m_parameters[U_LIGHTENABLED], 0);
750  }
751 
752  if (mesh->textureID > 0)
753  {
754  glUniform1i(m_parameters[U_COLOR_TEXTURE_ENABLED], 1);
755  glActiveTexture(GL_TEXTURE0);
756  glBindTexture(GL_TEXTURE_2D, mesh->textureID);
757  glUniform1i(m_parameters[U_COLOR_TEXTURE], 0);
758  }
759  else
760  {
761  glUniform1i(m_parameters[U_COLOR_TEXTURE_ENABLED], 0);
762  }
763  mesh->Render(); //this line should only be called once
764  if (mesh->textureID > 0)
765  {
766  glBindTexture(GL_TEXTURE_2D, 0);
767  }
768 }
